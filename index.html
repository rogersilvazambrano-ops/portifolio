package portfolio

import (
	"encoding/json"
	"fmt"
	"net/http"
	"time"
)

// Developer descreve o perfil profissional.
type Developer struct {
	Name      string
	Role      string
	Stack     []string
	Available bool
	Since     int
}

func NewDeveloper() *Developer {
	return &Developer{
		Name:      "Roger Zambrano",
		Role:      "Web Designer & Developer",
		Stack:     []string{"HTML", "CSS", "JS", "React"},
		Available: true,
		Since:     2021,
	}
}

func main() {
	dev := NewDeveloper()
	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		fmt.Fprintf(w, "> Olá, eu sou %s", dev.Name)
		json.NewEncoder(w).Encode(dev)
	})
	http.ListenAndServe(":8080", nil)
}
